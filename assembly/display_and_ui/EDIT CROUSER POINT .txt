.model small
.data
msg1      db       'MOHAMMAD AHMADI','$'   
.code
main  proc
    mov      ax, @data
    mov      ds, ax 
    call cls
    call printmsg1
    
    call chap 
      
    mov ax, 4c00h
    int 21h
main     endp 

cls proc 
    mov ah, 6h   ;cursor move
    mov al, 25   ;number of rows
    mov ch, 0
    mov cl, 0
    mov dh, 24   ;row
    mov dl, 79   ;column
    mov bh, 7    ;attribute
    int 10h
    ret
    cls endp

printmsg1 proc
    mov      ah, 2h   ;cursor move
    mov      dh, 16   ;row
    mov      dl, 30   ;column
    mov      bh, 0    ;page number
    int      10h
    lea dx, msg1
    mov ah, 9h
    int 21h     
    ret
    printmsg1 endp
end      main
