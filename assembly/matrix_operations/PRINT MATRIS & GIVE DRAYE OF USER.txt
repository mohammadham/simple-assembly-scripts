                         .MODEL SMALL  
                         
                         
PRINTSTR MACRO STR
MOV AH,09H
LEA DX,STR
INT 21H
ENDM   


GETSTR1 MACRO STR
LEA SI , STR
MOV DX , SI
MOV AH, 0AH
INT 21H
ENDM 

PRINTSTR1 MACRO STR

LEA si , STR
ADD si , 2
MOV DX , si 
MOV AH , 09H
INT 21H
ENDM
